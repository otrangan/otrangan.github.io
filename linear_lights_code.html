class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def sub(self, other):
        return Point(self.x - other.x, self.y - other.y)

def cross_product(u, v):
    return u.x * v.y - v.x * u.y

# orientation determinant
def orient(dots): 
    cc = cross_product(dots[1].sub(dots[0]), dots[2].sub(dots[0]))
    if cc == 0:
        return 0
    if cc < 0:
        return -1
    return 1

# euclidean square distance 
def dist_sq(p, q):
    return (p.x - q.x) ** 2 + (p.y - q.y) ** 2

# find intersection between two segments
def find_inter(dots):
    p1, p2, p3, p4 = dots[0], dots[1], dots[2], dots[3]
    
    if p1.x == p2.x and p3.x == p4.x:
        return None
        
    if p1.x - p2.x == 0:
        a2 = (p3.y - p4.y) / (p3.x - p4.x)
        b2 = p3.y - a2 * p3.x
        return Point(p1.x, a2 * p1.x + b2)

    if p3.x - p4.x == 0:
        a1 = (p1.y - p2.y) / (p1.x - p2.x);
        b1 = p1.y - a1 * p1.x;
        return Point(p3.x, a1 * p3.x + b1);

    a1 = (p1.y - p2.y) / (p1.x - p2.x);
    a2 = (p3.y - p4.y) / (p3.x - p4.x);
    b1 = p1.y - a1 * p1.x;
    b2 = p3.y - a2 * p3.x;
    
    if (a1 == a2):
        return None
    
    x = (b2 - b1) / (a1 - a2);
    y = a1 * x + b1;
    return Point(x,y)

import math
# count alpha function for vertex j regarding to light in c and previous vertex in p
def count_alpha(c, j, p):
    if (j == c):
        return 0
    
    sigma = orient([points[c], points[p], points[j]])
    
    a = points[j].sub(points[c])
    b = points[p].sub(points[c])    
    
    bb = math.atan2(b.y, b.x)
    aa = math.atan2(a.y, a.x)
    
    angle = bb - aa
    if angle < 0:
        angle += 2 * math.pi        
    if angle >= 2 * math.pi:
        angle -= 2 * math.pi
        
    angle = angle / math.pi / 2 * 360      
    if angle < 0:
        angle += 360            
    if angle > 180:
        angle = 360 - angle            

    r = alphas[p] + sigma * angle
    return r

def add_point(c, coords, prev):
    points.append(coords)

    a = count_alpha(c, pos, prev)
    alphas.append(a)

        W = (points[c], points[-1])
        if itr != None:
            add_point(c, itr, p)
        return linear_wait(W, c, j)

# implementation of pop procedure from the vertex j
def linear_pop(c, j):    
    while(len(stack) > 1):
        l, r = stack[-1], stack[-2]   
        if (alphas[l] >= alphas[j] and alphas[j] > alphas[r]):            
            stack.pop()

            y = find_inter([points[c], points[j], points[l], points[r]]) 
            add_point(c, y, r)

            k = (j + 1) % N
            if k == c:
                break
            if alphas[j] > alphas[k]:
                return linear_pop(c, k)            

            p = (j - 1 + N) % N
            left = -orient([points[p], points[j], points[k]])
            if left <= 0:                
                return linear_wait((points[j], points[stack[-1]]), c, k)
            return linear_push(c, j) 

        if (alphas[l] == alphas[r] and alphas[r] >= alphas[j]):
            p = (j - 1 + N) % N
            k = (j + 1) % N
            y = find_inter([points[p], points[j], points[l], points[r]])

            d1 = dist_sq(y, points[l])
            d2 = dist_sq(y, points[r])
            d = dist_sq(points[r], points[l])
            left = -orient([points[c], points[p], points[j]])
            
            if abs(sqrt(d) - (sqrt(d1) + sqrt(d2))) < 1e-5 and left <= 0:
                stack.pop()            
                if l == c:
                    return
                return linear_wait((points[stack[-1]], y), c, k)
        stack.pop()

# find visibility area for given light source in vertex c
def run(c):
    global alphas, stack, N, start
    alphas, stack, start = [], [], c
    W, N = None, len(points)

    for i in range(N):
        alphas.append(0)
    
    i = (c + 1) % N
    while i != c:    
        alphas[i] = count_alpha(c, i, (i - 1 + N)  % N)
        i = (i + 1) % N
        
    stack.append(c)    
    linear_push(c, (c + 1) % N)
